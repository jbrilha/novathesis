\newglossaryentry{LoRa}{
    name={LoRa},
    sort={LoRa},
    description={From "long range", proprietary radio communication technology}
}

\newglossaryentry{MQTT}{
    name={MQTT},
    sort={MQTT},
    description={Lightweight publish/subscribe messaging protocol designed for
        constrained devices and low-bandwidth networks. Previously stood for
        "Message Queuing Telemetry Transport", but since 2013 it does not stand
        for anything in particular
    }
}

\newglossaryentry{ubabel}{
    name={$\mu$-Babel},
    sort={ubabel},
    description={MicroBabel}
}

\newglossaryentry{RPi}{
    name={RPi},
    sort={rpi},
    description={Raspberry Pi full-size boards}
}

\newglossaryentry{hysteresis}{
    name={hysteresis},
    sort={hysteresis},
    description={Use of "decision inertia" or asymmetric thresholds to prevent
        frequent switching in response to small or short-lived changes in
        system metrics
    }
}

\newglossaryentry{ESP-NOW}{
    name={ESP-NOW},
    sort={ESP-NOW},
    description={Wireless communication protocol developed by Espressif.
        Enables direct and low-power communication between ESP-family devices
        without the need for a router.
    }
}

\newglossaryentry{ZigBee}{
    name={ZigBee},
    sort={ZigBee},
    description={Low-power, low-data rate wireless communication specification
        based on the IEEE 802.15.4 standard. Designed for short-range
        communication in mesh, star or tree network topologies.
    }
}

\newglossaryentry{Modbus}{
    name={Modbus},
    sort={Modbus},
    description={An application-layer client/server serial communication
        protocol, widely used in industrial automation systems.
    }
}
